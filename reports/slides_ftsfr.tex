\begin{frame}
  \frametitle{Overview}
  \begin{itemize}
  \item \alert{Motivation:}
  The set of forecasting methods is growing fast, especially machine-learning models. Do they actually improve forecasts of financial series?
  \item \alert{No Free Lunch:}
  The only way to know is to test methods on \alert{domain-specific (financial) data}, holding the data fixed across methods.
  \item \alert{Obstacle:}
  Each new method or paper brings its own data and its own cleaning, so comparisons entangle \alert{the forecasting method with the cleaning method}. Cleaning financial data also takes real domain expertise.
  \item \alert{Contribution:}
  FTSFR gathers canonical, domain-expert cleaning procedures for \alert{25 financial datasets in one place}, so the data stays fixed across 12 forecasting methods. Fully reproducible pipeline.
  \item \alert{Findings:}
  Returns stay near-unforecastable; \alert{basis spreads} reward hybrid/deep global models; bank \& intermediary indicators are competitive across families.
  \item \alert{Why it matters:}
  Change \alert{only} the cleaning convention and the leaderboard reshuffles, so fixed, well-cleaned data is a prerequisite for an honest comparison.
  \end{itemize}
\end{frame}

\section{Motivation}

\begin{frame}
\frametitle{Why Forecasting Matters in Finance}
\begin{itemize}
\item \alert{Asset pricing:} return predictability is the central empirical task---price variation reflects time-varying risk premia (Cochrane 2011; Kelly \& Xiu 2023)
\vspace{0.3cm}
\item \alert{Monetary \& macro policy:} central banks rely on factor models and large-panel forecasts of output, inflation, and financial conditions (Stock \& Watson 2002; McCracken 2016)
\vspace{0.3cm}
\item \alert{Supervisory stress testing:} multi-quarter projections of bank earnings, losses, and capital ratios calibrate buffers
\vspace{0.3cm}
\item \alert{Macroprudential surveillance:} early-warning indicators of systemic stress need lead time to act (Oet et al. 2011)
\vspace{0.3cm}
\item Across all of these the question is the same: \alert{given a panel of financial series, which forecasting method should one use?}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Many Methods, No Universal Winner}
\begin{itemize}
\item The toolkit of forecasting methods keeps growing, \alert{especially machine-learning models}
\item Competition winners keep changing as the methods evolve:
\begin{itemize}
  \item \alert{M3} (2018): classical statistical models dominate ML in accuracy \emph{and} cost
  \item \alert{M4} (2020): winner is a \emph{hybrid} RNN + exponential-smoothing system
  \item \alert{M5} (2022): swept by \emph{pure ML}, every entry beats the best statistical baselines
\end{itemize}
\item \alert{Monash} archive: \alert{no uniform winner} across its 25 datasets (Godahewa et al. 2021)
\end{itemize}
\vspace{0.2cm}
\begin{figure}
    \centering
    \includegraphics[width=0.65\textwidth]{\PathToAssets/m_competitions.png}
\end{figure}
\vspace{0.05cm}
\alert{No Free Lunch} (Wolpert \& Macready 1997): whether a new method really helps is an \alert{empirical, domain-specific question}, so we must test on financial data.
\end{frame}

\begin{frame}
\frametitle{The Obstacle: Each Study Brings Its Own Data}
\begin{itemize}
\item To learn whether a method genuinely helps, we must compare methods on the \alert{same fixed data.} But the same nominal series (``TRACE bonds,'' ``CRSP equities,'' ``G10 FX'') is built with different sources, filters, and sample windows in every study
\vspace{0.2cm}
\item \alert{Cleaning financial data takes domain expertise:} which instruments are actually tradable, which contracts are faulty, whether there is enough liquidity, and where the known data problems are
\vspace{0.2cm}
\item \alert{Concrete example (Dickerson et al. 2024):} bid--ask-averaged TRACE prices inject market-microstructure noise (MMN) into \emph{both} signal and realized return
\begin{itemize}
  \item MMN correction cuts the credit-spread long--short premium by \alert{50--65\%}
  \item Short-term reversal premium falls by \alert{over 90\%} (from $\sim$0.90\%/month to $\approx$0)
\end{itemize}
\vspace{0.2cm}
\item So a cleaning choice, not the method, can drive measured predictability. \alert{Comparisons across studies entangle the forecasting method with the cleaning method}
\end{itemize}
\end{frame}

  \begin{itemize}

  \begin{itemize}
    \item Each dataset uses the cleaning recipe from its canonical paper (tradability screens, faulty-contract removal, liquidity filters, known data fixes)

    \item Any researcher with WRDS / Bloomberg subscriptions reproduces each series exactly; pinned environments give consistent results across machines

\begin{frame}
  \frametitle{Dataset Selection: Intermediary Asset Pricing \& Canonical Cleaning}
  \begin{itemize}
  \item \alert{Theoretical foundation:} He, Kelly \& Manela (2017) intermediary asset pricing
  \begin{itemize}
    \item Retail investors access only stocks; intermediaries access exotic assets (CDS, options, bonds, commodities)
    \item Intermediary budget constraints create risk factors that price these assets---and these are exactly the markets critical for financial stability
  \end{itemize}
  \vspace{0.2cm}
  \item \alert{Basis spreads} gauge stress: when intermediaries face funding constraints, arbitrage gaps widen---critical early-warning indicators (following Siriwardane et al. 2021)
  \vspace{0.2cm}
  \item \alert{Methodological rigor:} each dataset uses the \alert{exact} cleaning methodology from the canonical paper for its asset class---no reinvention of procedures
  \begin{itemize}
    \item HKM intermediary factors, bank regulatory data, and yield curves add financial-stability coverage
    \item Established best practices, obtained from original authors where possible
  \end{itemize}
  \end{itemize}

\section{Methodology}

\begin{frame}{Forecasting Methods: 12 Models Across Four Families}
\begin{itemize}
\item \alert{Benchmark:} Historic Average (the hard-to-beat baseline for returns)
\vspace{0.2cm}
\item \alert{Classical Statistical:} ARIMA, SES (simple exponential smoothing family), Theta
\begin{itemize}
  \item Computationally efficient, interpretable, strong baselines
\end{itemize}
\vspace{0.2cm}
\item \alert{Hybrid (linear/decomposition):} DLinear, NLinear
\begin{itemize}
  \item Simple one-layer global models; DLinear decomposes trend/seasonal, NLinear normalizes inputs
\end{itemize}
\vspace{0.2cm}
\item \alert{Deep Learning (global):} DeepAR, N-BEATS, N-HiTS, Vanilla Transformer, TiDE, KAN
\begin{itemize}
  \item Train a \alert{single model across all series} in a dataset, borrowing strength cross-sectionally
\end{itemize}
\vspace{0.25cm}
\item All implemented via Nixtla \texttt{statsforecast}/\texttt{neuralforecast} with \alert{Auto} hyperparameter search on shared rolling-CV folds. \alert{Univariate, global, no exogenous regressors} (Godahewa et al. 2021).
\end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Evaluation Framework}
  \begin{itemize}
  \item \alert{Consistent preprocessing:} all models receive identical filtered datasets (short series removed; forward-fill, no look-ahead)
  \item \alert{Forecast design:}
    \begin{itemize}
      \item Horizon: 1 month (monthly), 21 trading / 30 calendar days (daily), 1 quarter (quarterly)
      \item Up to \alert{6 expanding rolling-origin CV windows}, reused by the Auto wrappers to keep tuning honest
    \end{itemize}
  \item \alert{Two complementary error metrics:}
  \[
  \mathrm{MASE}=\frac{\frac{1}{T}\sum_{t=1}^T |y_t-\hat y_t|}
  {\displaystyle \frac{1}{N-s}\sum_{t=s+1}^{N} |y_t - y_{t-s}|}
  \qquad
  R^2_{\text{oos}} = 1 - \frac{\sum (y_t-\hat y_t)^2}{\sum (y_t-\bar{y}_{\text{train}})^2}
  \]
  \begin{itemize}
    \item \alert{MASE} (Hyndman \& Koehler 2006): scale-free vs.\ seasonal naïve; robust to outliers; $<1$ beats naïve
    \item \alert{$R^2_{\text{oos}}$} (Campbell \& Thompson 2008): MSE reduction vs.\ historical average; standard in finance; $>0$ beats the mean
  \end{itemize}
  \end{itemize}
\end{frame}

\section{Results}

\begin{frame}
  \frametitle{Three Stylized Facts}
  \begin{enumerate}
  \item \alert{Asset returns stay near-unforecastable.}
  Even the best auto deep nets give $R^2_{\text{oos}}\approx 0$ across CRSP equities, TRACE corporate bonds, CDS, FX, and SPX options. The historical average remains the practical default.
  \\ \light{Caveat: univariate, global, no exogenous regressors---predictability may live in covariates we exclude.}
  \vspace{0.3cm}
  \item \alert{Basis spreads reward hybrid \& deep global models.}
  NHITS, NLinear, NBEATS, and TiDE all post \alert{median $R^2_{\text{oos}} > 0.64$}, well clear of the best classical contender (ARIMA, $0.53$), exploiting persistent seasonal funding-quarter structure.
  \vspace{0.3cm}
  \item \alert{Bank \& intermediary indicators are competitive.}
  Classical (ARIMA, Theta) and deep learning are roughly tied on \alert{median} $R^2_{\text{oos}}$; classical edges ahead on the \alert{mean} by avoiding deep models' occasional tail failures.
  \end{enumerate}
  \vspace{0.2cm}
  \centering \alert{The best model depends on the task---not on a single universal winner.}
\end{frame}

\begin{frame}
  \frametitle{Interpreting the Results: Metric and Segment Both Matter}
  \begin{itemize}
  \item \alert{Returns behave as theory predicts:} $R^2_{\text{oos}}\approx 0$ everywhere---the historical average is nearly unbeatable, consistent with efficient-markets intuition
  \vspace{0.25cm}
  \item \alert{Basis spreads are where modern methods earn their keep:} seasonal funding-quarter structure and global training let hybrid/deep models cut error sharply---the largest real gains in the benchmark
  \vspace{0.25cm}
  \item \alert{Supervisory / ``Other'' panels tilt classical-competitive:} Theta and ARIMA hold their own; classical methods avoid the tail failures that hit deep aggregates
  \vspace{0.25cm}
  \item \alert{Metric must match the question:}
  \begin{itemize}
    \item \textbf{MASE} for operational monitoring and seasonal accuracy (basis-spread surveillance)
    \item \textbf{$R^2_{\text{oos}}$} for return-risk questions benchmarked against the historical average
  \end{itemize}
  \vspace{0.25cm}
  \item \alert{Bottom line:} no universal best model---the appropriate default differs across return forecasting, basis-spread monitoring, and bank balance-sheet projection
  \end{itemize}
\end{frame}

\section{Sensitivity to Cleaning}

\begin{frame}
  \frametitle{Does the Cleaning Convention Move the Leaderboard? (TRACE)}
  \begin{itemize}
  \item \alert{Setup:} on the OSBAP corporate-bond panel, build credit-spread decile portfolios two ways---\alert{MMN-biased} vs.\ \alert{MMN-corrected} signal (Dickerson et al. 2024). \alert{Realized return, universe, and CV protocol held fixed.}
  \item Signals are 98\% rank-correlated, yet the long--short premium roughly \alert{doubles} (2.8\% $\to$ 5.6\%/yr)
  \end{itemize}
  \vspace{0.1cm}
  \centering
  \resizebox{0.62\textwidth}{!}{\input{\PathToOutput/forecasting/paper/mmn_sensitivity_tabular.tex}}

  \vspace{0.15cm}
  \footnotesize
  \alert{Leaderboard flips:} best model goes from \alert{KAN} ($R^2_{\text{oos}}{=}{+}0.284$) on the biased panel to \alert{NLinear} (${+}0.497$) on the corrected panel---same winner under both metrics, so not a metric artifact. Classical baselines are stable ($|\Delta R^2|\leq 0.04$) but never clear ${+}0.20$.
\end{frame}

\begin{frame}
  \frametitle{Sensitivity: The Takeaway}
  \begin{itemize}
  \item In every case we change \alert{only the cleaning convention}---same source, same realized returns---yet the forecasting leaderboard \alert{reshuffles materially}
  \vspace{0.3cm}
  \item The volatility is concentrated in exactly the \alert{deep models a practitioner would reach for}; classical baselines are stable but rarely the best available forecast
  \vspace{0.3cm}
  \item \alert{Implication:} any cross-paper comparison that does not hold the cleaning recipe fixed \alert{confounds the forecasting method with the cleaning method}
  \vspace{0.3cm}
  \item Even coarse summaries (``which model gives positive $R^2_{\text{oos}}$?'') are cleaning-dependent
  \vspace{0.3cm}
  \item \alert{This is why holding the data fixed, and cleaning it with domain expertise, is a prerequisite for comparing methods}, and why reproducible pipelines like FTSFR are the value we provide
  \end{itemize}
\end{frame}

\section{Conclusions}

\begin{frame}
  \frametitle{Contributions \& Future Work}
  \begin{itemize}
  \item \alert{Main contributions:}
  \begin{itemize}
    \item First comprehensive, \alert{fixed-data} financial time series forecasting benchmark (25 datasets, canonical cleaning, 12 methods)
    \item Three stylized facts: returns near-unforecastable; basis spreads reward hybrid/deep global models; bank/intermediary indicators competitive across families
    \item A sensitivity exercise showing method rankings are \alert{entangled with cleaning choices}---resolved only by holding the data fixed
    \item Open-source pipeline enabling exact, reproducible replication
  \end{itemize}
  \vspace{0.3cm}
  \item \alert{For practitioners:} match the default to the task---classical baselines for returns, hybrid/deep global models for funding-stress and basis-spread monitoring
  \vspace{0.3cm}
  \item \alert{Future work:}
  \begin{itemize}
    \item Admit exogenous covariates and multivariate / global-multivariate model classes
    \item Multi-step and long-horizon forecasting; integration with stress-testing frameworks
    \item Expand coverage to international markets
  \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Model Deep Dive: Classical \& Hybrid}
\begin{itemize}
    \item \alert{Classical Statistical Methods}
    \begin{itemize}
        \item \textbf{Auto wrappers:} Theta, ARIMA, SES reuse rolling-CV folds to tune parameters and avoid overfitting
        \item \textbf{Theta:} decomposes series by modifying local curvature into short/long-term components, then extrapolates separately---simple but often beats sophisticated methods on short or noisy series
        \item \textbf{ARIMA:} mixed performance but generally beats simple baselines
        \item \textbf{SES:} exponential weighting of past observations
    \end{itemize}
    \vspace{0.3cm}
    \item \alert{Hybrid Linear Models}
    \begin{itemize}
        \item \textbf{DLinear:} seasonal-trend decomposition with linear layers
        \item \textbf{NLinear:} normalization trick to handle distribution shifts
        \item Surprisingly effective; global training lets them borrow strength across series
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Model Deep Dive: Deep Learning (Global Methods)}
    \begin{itemize}
        \item \textbf{Global Auto models} (DeepAR, NBEATS, NHITS, TiDE, KAN, Transformer) train one model across all series and tune via Bayesian search on shared validation folds
        \vspace{0.3cm}
        \item \textbf{DeepAR:} RNN with LSTM cells; probabilistic forecasts
        \item \textbf{Transformer:} self-attention without recurrence; effectiveness for long horizons debated---simple linear models often competitive
        \item \textbf{TiDE:} MLP encoder-decoder, 5--10x faster than Transformers; handles non-linear dependencies
        \item \textbf{N-BEATS:} deep architecture with residual links; interpretable via trend/seasonality decomposition
        \item \textbf{N-HiTS:} extends N-BEATS for long horizons via hierarchical interpolation and multi-rate sampling
        \item \textbf{KAN:} Kolmogorov-Arnold networks with spline-based activations
    \end{itemize}
\end{frame}

  \item Security-level (100s--1000s) for global forecasting

\begin{frame}
  \frametitle{Technical Implementation Details}
  \label{slide:technical}
\begin{itemize}
\item \alert{Forecasting packages:} StatsForecast \& NeuralForecast (Nixtla)---standardized, reproducible implementations
\vspace{0.3cm}
\item \alert{Data preprocessing:}
\begin{itemize}
  \item Forward-fill for missing values (no look-ahead bias)
  \item Length-based filtering with dataset-specific thresholds (minimum test coverage matches horizon)
  \item Frequency-specific horizons: 30-day (calendar), 21-day (trading), 1-month, 1-quarter; up to six rolling windows per dataset
\end{itemize}
\vspace{0.3cm}
\item \alert{Evaluation:}
\begin{itemize}
  \item Rolling-origin cross-validation aligned across statistical and neural models
  \item Two complementary error metrics (MASE, $R^2_{\text{oos}}$) for robust comparison
\end{itemize}
\end{itemize}
